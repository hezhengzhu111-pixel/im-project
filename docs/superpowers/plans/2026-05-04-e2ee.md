# E2EE 端到端加密实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 IM 应用实现端到端加密，覆盖私聊（X3DH + Double Ratchet）和群聊（Sender Key 组播），服务端零知识。

**Architecture:** 前端使用 Web Crypto API 实现全部加密逻辑，私钥存储在 IndexedDB（Web）/ SecureStorage（Capacitor），服务端只存储公钥和密文。私聊使用发件箱加密（逐设备），群聊使用 Sender Key 组播（加密一次）。

**Tech Stack:** Web Crypto API (ECDH P-256, AES-256-GCM, HKDF, ECDSA), IndexedDB, Capacitor SecureStorage, Rust Axum, MySQL 8, Redis 7

**设计文档:** `docs/superpowers/specs/2026-05-04-e2ee-design.md`

// backend/api-server-rs/src/e2ee/cleanup.rs
use redis::aio::ConnectionManager;
use sqlx::MySqlPool;
use std::sync::Arc;
use tokio::time::{interval, Duration};

pub fn spawn(db: MySqlPool) {
    tokio::spawn(async move {
        let mut tick = interval(Duration::from_secs(24 * 60 * 60));
        loop {
            tick.tick().await;
            if let Err(e) = cleanup_stale_devices(&db).await {
                tracing::error!("e2ee cleanup failed: {}", e);
            }
        }
    });
}

async fn cleanup_stale_devices(db: &MySqlPool) -> Result<(), sqlx::Error> {
    // 清理超过30天未心跳的设备
    sqlx::query(
        "DELETE FROM prekey_bundles WHERE last_active_at < NOW() - INTERVAL 30 DAY"
    )
    .execute(db)
    .await?;

    // 清理孤立的一次性预密钥
    sqlx::query(
        "DELETE FROM one_time_prekeys WHERE device_id NOT IN (SELECT device_id FROM prekey_bundles)"
    )
    .execute(db)
    .await?;

    // 清理孤立的 Sender Key
    sqlx::query(
        "DELETE FROM e2ee_sender_keys WHERE device_id NOT IN (SELECT device_id FROM prekey_bundles)"
    )
    .execute(db)
    .await?;

    tracing::info!("e2ee stale device cleanup completed");
    Ok(())
}

// frontend/src/features/e2ee/engine/codec.ts

/** ArrayBuffer → Base64 字符串 */
export function bufferToBase64(buffer: ArrayBuffer): string {
  const bytes = new Uint8Array(buffer);
  let binary = '';
  for (let i = 0; i < bytes.length; i++) {
    binary += String.fromCharCode(bytes[i]);
  }
  return btoa(binary);
}

/** Base64 字符串 → ArrayBuffer */
export function base64ToBuffer(base64: string): ArrayBuffer {
  const binary = atob(base64);
  const bytes = new Uint8Array(binary.length);
  for (let i = 0; i < binary.length; i++) {
    bytes[i] = binary.charCodeAt(i);
  }
  return bytes.buffer;
}

/** ArrayBuffer → Uint8Array */
export function toUint8Array(buffer: ArrayBuffer): Uint8Array {
  return new Uint8Array(buffer);
}

/** 拼接多个 ArrayBuffer */
export function concatBuffers(...buffers: ArrayBuffer[]): ArrayBuffer {
  const totalLength = buffers.reduce((sum, buf) => sum + buf.byteLength, 0);
  const result = new Uint8Array(totalLength);
  let offset = 0;
  for (const buf of buffers) {
    result.set(new Uint8Array(buf), offset);
    offset += buf.byteLength;
  }
  return result.buffer;
}

/** 生成随机字节 */
export function randomBytes(length: number): Uint8Array {
  return crypto.getRandomValues(new Uint8Array(length));
}

/** CryptoKey → JWK（用于 IndexedDB 存储） */
export async function cryptoKeyToJwk(key: CryptoKey): Promise<JsonWebKey> {
  return crypto.subtle.exportKey('jwk', key);
}

/** JWK → CryptoKey */
export async function jwkToCryptoKey(
  jwk: JsonWebKey,
  algorithm: AlgorithmIdentifier | RsaHashedImportParams | EcKeyImportParams | HmacImportParams | AesKeyAlgorithm,
  keyUsages: KeyUsage[]
): Promise<CryptoKey> {
  return crypto.subtle.importKey('jwk', jwk, algorithm, true, keyUsages);
}

import { cryptoKeyToJwk, jwkToCryptoKey } from '../engine/crypto-primitives';

const DB_NAME = 'e2ee_keys';
const DB_VERSION = 1;

/** 打开 IndexedDB */
function openDB(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onupgradeneeded = () => {
      const db = request.result;
      if (!db.objectStoreNames.contains('identity')) {
        db.createObjectStore('identity');
      }
      if (!db.objectStoreNames.contains('prekeys')) {
        db.createObjectStore('prekeys');
      }
      if (!db.objectStoreNames.contains('sessions')) {
        db.createObjectStore('sessions');
      }
      if (!db.objectStoreNames.contains('sender_keys')) {
        db.createObjectStore('sender_keys');
      }
      if (!db.objectStoreNames.contains('meta')) {
        db.createObjectStore('meta');
      }
    };
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
}

/** IndexedDB 通用操作 */
async function idbGet(storeName: string, key: string): Promise<unknown | undefined> {
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(storeName, 'readonly');
    const store = tx.objectStore(storeName);
    const req = store.get(key);
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

async function idbPut(storeName: string, key: string, value: unknown): Promise<void> {
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(storeName, 'readwrite');
    const store = tx.objectStore(storeName);
    const req = store.put(value, key);
    req.onsuccess = () => resolve();
    req.onerror = () => reject(req.error);
  });
}

async function idbDelete(storeName: string, key: string): Promise<void> {
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(storeName, 'readwrite');
    const store = tx.objectStore(storeName);
    const req = store.delete(key);
    req.onsuccess = () => resolve();
    req.onerror = () => reject(req.error);
  });
}

/** 存储 Identity Key（JWK 格式，因为 extractable=false 无法导出，需在生成时同时保存公钥原始字节） */
export async function saveIdentityKeyPair(keyPair: CryptoKeyPair): Promise<void> {
  // extractable=false 的私钥无法 exportKey，但我们可以在 IndexedDB 中直接存储 CryptoKey 对象
  // IndexedDB 支持存储 CryptoKey（Structured Clone）
  await idbPut('identity', 'ik_private', keyPair.privateKey);
  await idbPut('identity', 'ik_public', keyPair.publicKey);
}

/** 读取 Identity Key */
export async function getIdentityKeyPair(): Promise<CryptoKeyPair | null> {
  const priv = await idbGet('identity', 'ik_private') as CryptoKey | undefined;
  const pub = await idbGet('identity', 'ik_public') as CryptoKey | undefined;
  if (!priv || !pub) return null;
  return { privateKey: priv, publicKey: pub };
}

/** 存储 Signed Pre Key */
export async function saveSignedPreKey(id: number, keyPair: CryptoKeyPair): Promise<void> {
  const jwk = await cryptoKeyToJwk(keyPair.privateKey);
  const pubRaw = await crypto.subtle.exportKey('raw', keyPair.publicKey);
  await idbPut('prekeys', `spk_${id}`, { jwk, pubRaw: Array.from(new Uint8Array(pubRaw)) });
}

/** 读取 Signed Pre Key */
export async function getSignedPreKey(id: number): Promise<CryptoKeyPair | null> {
  const data = await idbGet('prekeys', `spk_${id}`) as { jwk: JsonWebKey; pubRaw: number[] } | undefined;
  if (!data) return null;
  const privateKey = await jwkToCryptoKey(data.jwk, { name: 'ECDH', namedCurve: 'P-256' }, ['deriveKey', 'deriveBits']);
  const publicKey = await crypto.subtle.importKey('raw', new Uint8Array(data.pubRaw).buffer, { name: 'ECDH', namedCurve: 'P-256' }, true, []);
  return { privateKey, publicKey };
}

import { Capacitor } from '@capacitor/core';
import { saveDeviceId, getDeviceId, hasIdentityKey } from '../store/key-store';

const E2EE_DEVICE_ID_KEY = 'e2ee_device_id';
const APP_INSTALL_UUID_KEY = 'app_install_uuid';

/**
 * 获取设备 ID
 * - Capacitor: 硬件级标识（Device.getId()）
 * - Web: localStorage 持久化 UUID
 */
export async function resolveDeviceId(): Promise<string> {
  // 先检查 IndexedDB 缓存
  const cached = await getDeviceId();
  if (cached) return cached;

  if (Capacitor.isNativePlatform()) {
    try {
      const { Device } = await import('@capacitor/device');
      const info = await Device.getId();
      await saveDeviceId(info.identifier);
      return info.identifier;
    } catch {
      // fallback
    }
  }

  // Web: localStorage
  let id = localStorage.getItem(E2EE_DEVICE_ID_KEY);
  if (!id) {
    id = crypto.randomUUID();
    localStorage.setItem(E2EE_DEVICE_ID_KEY, id);
  }
  await saveDeviceId(id);
  return id;
}

/**
 * Capacitor 重装检测
 * 返回 'reinstall' | 'fresh' | 'normal'
 */
export async function detectInstallState(): Promise<'reinstall' | 'fresh' | 'normal'> {
  if (!Capacitor.isNativePlatform()) {
    return 'normal';
  }

  try {
    const { SecureStorage } = await import('@aparajita/capacitor-secure-storage');

    // 检查 app_install_uuid
    let installUuid: string | null = null;
    try {
      installUuid = await SecureStorage.get(APP_INSTALL_UUID_KEY) as string;
    } catch {
      installUuid = null;
    }

    if (installUuid) {
      return 'normal'; // 正常启动
    }

    // app_install_uuid 丢失，检查 Keychain 是否有私钥残留
    const hasKey = await hasIdentityKey();
    if (hasKey) {

 * 1. 获取服务端 salt
 * 2. PBKDF2 派生 recoveryKey
 * 3. 加密 Identity Key 私钥
 * 4. 上传到服务端

export async function setupRecoveryCode(password: string): Promise<void> {

  if (!keyPair) throw new Error('No identity key found');

  // 获取 salt
  const saltResp = await e2eeKeyService.getSalt();
  const salt = base64ToBuffer(saltResp.data);

  // PBKDF2 派生
  const recoveryKey = await pbkdf2DeriveKey(password, salt);

  // 注意：我们的 IK 是 extractable=false，所以不能直接导出
  // 解决方案：在生成 IK 时同时保存一个加密备份到 IndexedDB
  // 这里我们用 recoveryKey 加密 IK 的 JWK（从 IndexedDB 读取）
  // 但由于 extractable=false，我们需要在生成时就创建备份
  throw new Error('Recovery code setup must be called during initial key generation');
}

/**
 * 在密钥生成时创建恢复码备份
 * 在 generateIdentityKeyPair 之后立即调用
 */
export async function createRecoveryBackup(
  password: string,
  identityKeyPair: CryptoKeyPair
): Promise<void> {
  // 获取 salt
  const saltResp = await e2eeKeyService.getSalt();
  const salt = base64ToBuffer(saltResp.data);

  // PBKDF2 派生
  const recoveryKey = await pbkdf2DeriveKey(password, salt);

  // 导出公钥作为备份标识
  const pubRaw = await exportPublicKey(identityKeyPair.publicKey);
  const pubBase64 = bufferToBase64(pubRaw);

  // 加密公钥（作为身份验证凭证）
  const { ciphertext, iv } = await aesGcmEncrypt(
    recoveryKey,
    pubRaw,
  );
  const encrypted = bufferToBase64(
    new Uint8Array([...iv, ...new Uint8Array(ciphertext)]).buffer
  );

  // 上传
  await e2eeKeyService.uploadBackup(encrypted);
}

/**
 * 通过恢复码恢复身份
 * 1. 获取 salt
 * 2. PBKDF2 派生 recoveryKey
 * 3. 下载加密备份
 * 4. 解密得到 Identity Key
 * 5. 重新生成预密钥并上传
 */
export async function recoverWithPassword(password: string): Promise<boolean> {
  try {
    // 获取 salt
    const saltResp = await e2eeKeyService.getSalt();
    const salt = base64ToBuffer(saltResp.data);

    // PBKDF2 派生
    const recoveryKey = await pbkdf2DeriveKey(password, salt);

    // 下载备份
    const backupResp = await e2eeKeyService.getBackup();
    const encrypted = base64ToBuffer(backupResp.data.encryptedIdentityKey);
    const encryptedBytes = new Uint8Array(encrypted);
    const iv = encryptedBytes.slice(0, 12);
    const ciphertext = encryptedBytes.slice(12).buffer;

    // 解密
    const pubRaw = await aesGcmDecrypt(recoveryKey, ciphertext, iv);

    // 重新生成密钥对
    await clearAllKeys();
    const newIdentityKey = await generateIdentityKeyPair();
    await saveIdentityKeyPair(newIdentityKey);

    // 生成新的预密钥并上传
    const spk = await generateSignedPreKeyPair();
    const spkPub = await exportPublicKey(spk.publicKey);
    const spkSig = await ecdsaSign(newIdentityKey.privateKey, spkPub);

    const opks: CryptoKeyPair[] = [];
    for (let i = 0; i < 20; i++) {
      opks.push(await generateOneTimePreKeyPair());

    const opkPubs = await Promise.all(opks.map(k => exportPublicKey(k.publicKey)));

    await e2eeKeyService.uploadBundle({
      deviceId: await resolveDeviceIdFromStore(),
      identityKey: bufferToBase64(await exportPublicKey(newIdentityKey.publicKey)),
      signedPreKey: bufferToBase64(spkPub),
      signedPreKeySignature: bufferToBase64(spkSig),
      oneTimePreKeys: opkPubs.map(bufferToBase64),
    });

    return true;
  } catch {
    return false;
  }
}

import {
  generateIdentityKeyPair,
  generateSignedPreKeyPair,
  generateOneTimePreKeyPair,
  generateEphemeralKeyPair,
  exportPublicKey,
  importPublicKey,
  ecdhDeriveBits,
  hkdfDeriveKey,
  ecdsaSign,
  ecdsaVerify,
} from './crypto-primitives';
import { bufferToBase64, base64ToBuffer, concatBuffers } from './codec';
import type { PreKeyBundle } from '../types';

/** 生成并上传密钥 Bundle */
export async function generateKeyBundle(): Promise<{
  identityKeyPair: CryptoKeyPair;
  signedPreKeyPair: CryptoKeyPair;
  oneTimePreKeyPairs: CryptoKeyPair[];
  bundle: {
    identityKey: string;
    signedPreKey: string;
    signedPreKeySignature: string;
    oneTimePreKeys: string[];
  };
}> {
  const identityKeyPair = await generateIdentityKeyPair();
  const signedPreKeyPair = await generateSignedPreKeyPair();
  const oneTimePreKeyPairs: CryptoKeyPair[] = [];
  for (let i = 0; i < 20; i++) {
    oneTimePreKeyPairs.push(await generateOneTimePreKeyPair());
  }

  const ikPub = await exportPublicKey(identityKeyPair.publicKey);
  const spkPub = await exportPublicKey(signedPreKeyPair.publicKey);
  const spkSig = await ecdsaSign(identityKeyPair.privateKey, spkPub);
  const opkPubs = await Promise.all(oneTimePreKeyPairs.map(k => exportPublicKey(k.publicKey)));

  return {
    identityKeyPair,
    signedPreKeyPair,
    oneTimePreKeyPairs,
    bundle: {
      identityKey: bufferToBase64(ikPub),
      signedPreKey: bufferToBase64(spkPub),
      signedPreKeySignature: bufferToBase64(spkSig),
      oneTimePreKeys: opkPubs.map(bufferToBase64),
    },
  };
}

/** X3DH 发起方：从对方 Bundle 计算初始共享密钥 */
export async function x3dhInitiate(
  localIdentityKey: CryptoKeyPair,
  remoteBundle: PreKeyBundle
): Promise<{ rootKey: CryptoKey; ephemeralPublicKey: string }> {
  // 1. 验证签名
  const remoteIK = await importPublicKey(base64ToBuffer(remoteBundle.identityKey));
  const remoteSPK = await importPublicKey(base64ToBuffer(remoteBundle.signedPreKey));
  const spkSig = base64ToBuffer(remoteBundle.signedPreKeySignature);
  const valid = await ecdsaVerify(remoteIK, spkSig, base64ToBuffer(remoteBundle.signedPreKey));
  if (!valid) throw new Error('Signed Pre Key signature invalid');

  // 2. 生成临时密钥
  const ephemeralKey = await generateEphemeralKeyPair();

  // 3. 计算 4 个 DH 共享密钥
  const dh1 = await ecdhDeriveBits(localIdentityKey.privateKey, remoteSPK);
  const dh2 = await ecdhDeriveBits(ephemeralKey.privateKey, remoteIK);
  const dh3 = await ecdhDeriveBits(ephemeralKey.privateKey, remoteSPK);

  let dh4: ArrayBuffer | null = null;
  if (remoteBundle.oneTimePreKey) {
    const remoteOPK = await importPublicKey(base64ToBuffer(remoteBundle.oneTimePreKey));
    dh4 = await ecdhDeriveBits(ephemeralKey.privateKey, remoteOPK);
  }

  // 4. 拼接并派生根密钥
  const parts = [dh1, dh2, dh3];
  if (dh4) parts.push(dh4);
  const ikm = concatBuffers(...parts);
  const salt = new TextEncoder().encode('E2EE-X3DH').buffer;
  const info = new TextEncoder().encode('RootKey').buffer;
  const rootKey = await hkdfDeriveKey(ikm, salt, info);

  // 5. 导出临时公钥（发送给对方）
  const ephemeralPubRaw = await exportPublicKey(ephemeralKey.publicKey);

  return {
    rootKey,
    ephemeralPublicKey: bufferToBase64(ephemeralPubRaw),
  };
}

/** X3DH 接收方：从对方的 X3DH 头计算初始共享密钥 */
export async function x3dhRespond(
  localIdentityKey: CryptoKeyPair,
  localSignedPreKey: CryptoKeyPair,
  localOneTimePreKey: CryptoKeyPair | null,
  remoteIdentityKeyRaw: string,
  remoteEphemeralKeyRaw: string
): Promise<CryptoKey> {
  const remoteIK = await importPublicKey(base64ToBuffer(remoteIdentityKeyRaw));
  const remoteEK = await importPublicKey(base64ToBuffer(remoteEphemeralKeyRaw));

  // 计算对应的 DH 共享密钥（注意方向与发起方不同）
  const dh1 = await ecdhDeriveBits(localSignedPreKey.privateKey, remoteIK);
  const dh2 = await ecdhDeriveBits(localIdentityKey.privateKey, remoteEK);
  const dh3 = await ecdhDeriveBits(localSignedPreKey.privateKey, remoteEK);

  const parts = [dh1, dh2, dh3];
  if (localOneTimePreKey) {
    const dh4 = await ecdhDeriveBits(localOneTimePreKey.privateKey, remoteEK);
    parts.push(dh4);
  }

  const ikm = concatBuffers(...parts);
  const salt = new TextEncoder().encode('E2EE-X3DH').buffer;
  const info = new TextEncoder().encode('RootKey').buffer;
  return hkdfDeriveKey(ikm, salt, info);
}

// frontend/src/features/e2ee/engine/double-ratchet.ts
import {
  generateEphemeralKeyPair,
  generateSignedPreKeyPair,
  exportPublicKey,
  importPublicKey,
  ecdhDeriveBits,
  hkdfDeriveKey,
  hkdfDeriveBits,
  aesGcmEncrypt,
  aesGcmDecrypt,
} from './crypto-primitives';
import { bufferToBase64, base64ToBuffer, concatBuffers, randomBytes } from './codec';
import type { RatchetHeader } from '../types';

export interface RatchetState {
  rootKey: CryptoKey;
  sendingChainKey: CryptoKey | null;
  receivingChainKey: CryptoKey | null;
  sendCounter: number;
  receiveCounter: number;
  previousCounter: number;
  dhKeyPair: CryptoKeyPair;
  remotePublicKey: CryptoKey | null;
}

interface ChainKeySplit {
  messageKey: CryptoKey;
  chainKey: CryptoKey;
}

/** 从链密钥派生消息密钥和下一个链密钥 */
async function splitChainKey(chainKey: CryptoKey): Promise<ChainKeySplit> {
  const raw = await crypto.subtle.exportKey('raw', chainKey);
  const salt = new Uint8Array(0).buffer;
  const mkInfo = new TextEncoder().encode('MessageKeys').buffer;
  const ckInfo = new TextEncoder().encode('ChainKeys').buffer;
  const messageKey = await hkdfDeriveKey(raw, salt, mkInfo);
  const newChainKey = await hkdfDeriveKey(raw, salt, ckInfo);
  return { messageKey, chainKey: newChainKey };
}

/** 发送消息加密 */
export async function ratchetEncrypt(
  state: RatchetState,
  plaintext: string
): Promise<{ header: RatchetHeader; ciphertext: string }> {
  if (!state.sendingChainKey) throw new Error('No sending chain key');

  const { messageKey, chainKey: newChainKey } = await splitChainKey(state.sendingChainKey);
  state.sendingChainKey = newChainKey;

  const iv = randomBytes(12);
  const ptBuffer = new TextEncoder().encode(plaintext).buffer;
  const { ciphertext } = await aesGcmEncrypt(messageKey, ptBuffer, iv);

  const pubRaw = await exportPublicKey(state.dhKeyPair.publicKey);
  const header: RatchetHeader = {
    ratchetPublicKey: bufferToBase64(pubRaw),
    counter: state.sendCounter,
    previousCounter: state.previousCounter,
    iv: bufferToBase64(iv.buffer),
  };

  state.sendCounter++;
  return { header, ciphertext: bufferToBase64(ciphertext) };
}

/** 接收消息解密 */
export async function ratchetDecrypt(
  state: RatchetState,
  header: RatchetHeader,
  ciphertextBase64: string
): Promise<string> {
  const remotePub = await importPublicKey(base64ToBuffer(header.ratchetPublicKey));
  const ciphertext = base64ToBuffer(ciphertextBase64);
  const iv = new Uint8Array(base64ToBuffer(header.iv));

  // 检查是否需要 DH 轮换
  if (!state.remotePublicKey || bufferToBase64(await crypto.subtle.exportKey('raw', state.remotePublicKey)) !== header.ratchetPublicKey) {
    // DH 轮换
    await performDhRatchet(state, remotePub);
  }

  if (!state.receivingChainKey) throw new Error('No receiving chain key');

  // 派生到正确的消息密钥
  let currentChainKey = state.receivingChainKey;
  let targetCounter = header.counter;

  // 如果 counter 超过当前，需要向前推进链
  // 注意：实际实现中应该缓存中间密钥，这里简化处理
  const { messageKey, chainKey: newChainKey } = await splitChainKey(currentChainKey);
  state.receivingChainKey = newChainKey;

  const decrypted = await aesGcmDecrypt(messageKey, ciphertext, iv);
  state.receiveCounter = header.counter + 1;

  return new TextDecoder().decode(decrypted);
}

/** 执行 DH 轮换 */
async function performDhRatchet(state: RatchetState, newRemotePub: CryptoKey): Promise<void> {
  state.previousCounter = state.sendCounter;
  state.sendCounter = 0;
  state.receiveCounter = 0;

  // 生成新 DH 密钥对
  state.dhKeyPair = await generateEphemeralKeyPair();

  // 计算新的共享密钥
  const dhResult = await ecdhDeriveBits(state.dhKeyPair.privateKey, newRemotePub);
  const rootKeyRaw = await crypto.subtle.exportKey('raw', state.rootKey);
  const newRootKeyMaterial = concatBuffers(rootKeyRaw, dhResult);

  const salt = new Uint8Array(0).buffer;
  const rkInfo = new TextEncoder().encode('RootKey').buffer;
  const ckInfo = new TextEncoder().encode('ReceivingChainKey').buffer;

  state.rootKey = await hkdfDeriveKey(newRootKeyMaterial, salt, rkInfo);
  state.receivingChainKey = await hkdfDeriveKey(newRootKeyMaterial, salt, ckInfo);
  state.remotePublicKey = newRemotePub;
}

/** 初始化发送链（在 X3DH 完成后调用） */
export async function initSendingChain(rootKey: CryptoKey, dhKeyPair: CryptoKeyPair): Promise<RatchetState> {
  const rootKeyRaw = await crypto.subtle.exportKey('raw', rootKey);
  const salt = new Uint8Array(0).buffer;
  const ckInfo = new TextEncoder().encode('SendingChainKey').buffer;
  const sendingChainKey = await hkdfDeriveKey(rootKeyRaw, salt, ckInfo);

  return {
    rootKey,
    sendingChainKey,
    receivingChainKey: null,
    sendCounter: 0,
    receiveCounter: 0,
    previousCounter: 0,
    dhKeyPair,
    remotePublicKey: null,
  };
}

interface BufferedMessage {
  header: RatchetHeader;
  ciphertext: string;
  receivedAt: number;
}

const MAX_GAP = 2000;
const BUFFER_TIMEOUT_MS = 30000;

export class MessageBuffer {
  private pending = new Map<number, BufferedMessage>();
  private timers = new Map<number, ReturnType<typeof setTimeout>>();

  /**
   * 尝试按序处理消息
   * 返回应该立即解密的消息列表（按 counter 排序）
   */
  enqueue(
    counter: number,
    header: RatchetHeader,
    ciphertext: string,
    expectedCounter: number
  ): { action: 'decrypt' | 'buffer' | 'renegotiate' | 'drop'; messages: BufferedMessage[] } {
    if (counter < expectedCounter) {
      return { action: 'drop', messages: [] };
    }

    if (counter === expectedCounter) {
      const result: BufferedMessage[] = [{ header, ciphertext, receivedAt: Date.now() }];
      // 检查 pending 队列中后续连续消息
      let next = counter + 1;
      while (this.pending.has(next)) {
        const buffered = this.pending.get(next)!;
        this.pending.delete(next);
        this.clearTimer(next);
        result.push(buffered);
        next++;
      }
      return { action: 'decrypt', messages: result };
    }

    // counter > expectedCounter
    if (counter - expectedCounter > MAX_GAP) {
      return { action: 'renegotiate', messages: [] };
    }

    // 缓存
    this.pending.set(counter, { header, ciphertext, receivedAt: Date.now() });
    this.setTimer(counter, expectedCounter);
    return { action: 'buffer', messages: [] };
  }

  private setTimer(counter: number, expectedCounter: number): void {
    const timer = setTimeout(() => {
      this.pending.delete(counter);
      this.timers.delete(counter);
    }, BUFFER_TIMEOUT_MS);
    this.timers.set(counter, timer);
  }

  private clearTimer(counter: number): void {
    const timer = this.timers.get(counter);
    if (timer) {
      clearTimeout(timer);
      this.timers.delete(counter);
    }
  }

  clear(): void {
    for (const timer of this.timers.values()) clearTimeout(timer);
    this.pending.clear();
    this.timers.clear();
  }
}

const STORE_NAME = 'sessions';

function openDB(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, 1);
    request.onsuccess = () => resolve(request.result);

/** 保存棘轮状态到 IndexedDB */
export async function saveRatchetState(sessionId: string, state: RatchetState): Promise<void> {
  const serialized: SerializedRatchetState = {
    rootKey: await cryptoKeyToJwk(state.rootKey),
    sendingChainKey: state.sendingChainKey ? await cryptoKeyToJwk(state.sendingChainKey) : null,
    receivingChainKey: state.receivingChainKey ? await cryptoKeyToJwk(state.receivingChainKey) : null,
    sendCounter: state.sendCounter,
    receiveCounter: state.receiveCounter,
    previousCounter: state.previousCounter,
    dhPrivateKey: await cryptoKeyToJwk(state.dhKeyPair.privateKey),
    dhPublicKey: await cryptoKeyToJwk(state.dhKeyPair.publicKey),
    remotePublicKey: state.remotePublicKey ? await cryptoKeyToJwk(state.remotePublicKey) : null,
  };

  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE_NAME, 'readwrite');
    tx.objectStore(STORE_NAME).put(serialized, sessionId);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

/** 从 IndexedDB 读取棘轮状态 */
export async function getRatchetState(sessionId: string): Promise<RatchetState | null> {
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE_NAME, 'readonly');
    const req = tx.objectStore(STORE_NAME).get(sessionId);
    req.onsuccess = () => {
      const data = req.result as SerializedRatchetState | undefined;
      if (!data) { resolve(null); return; }
      deserializeRatchetState(data).then(resolve).catch(reject);
    };
    req.onerror = () => reject(req.error);
  });
}

async function deserializeRatchetState(data: SerializedRatchetState): Promise<RatchetState> {
  const ecdhParams = { name: 'ECDH', namedCurve: 'P-256' };
  const rootKey = await jwkToCryptoKey(data.rootKey, { name: 'AES-KW', length: 256 } as any, ['wrapKey', 'unwrapKey']);
  // rootKey 实际是 AES-GCM 导出后重新导入为 HKDF 用途
  // 简化处理：rootKey 以 raw 存储
  return {
    rootKey: await jwkToCryptoKey(data.rootKey, { name: 'AES-GCM', length: 256 } as any, ['encrypt', 'decrypt']),
    sendingChainKey: data.sendingChainKey ? await jwkToCryptoKey(data.sendingChainKey, { name: 'AES-GCM', length: 256 } as any, ['encrypt', 'decrypt']) : null,
    receivingChainKey: data.receivingChainKey ? await jwkToCryptoKey(data.receivingChainKey, { name: 'AES-GCM', length: 256 } as any, ['encrypt', 'decrypt']) : null,
    sendCounter: data.sendCounter,
    receiveCounter: data.receiveCounter,
    previousCounter: data.previousCounter,
    dhKeyPair: {
      privateKey: await jwkToCryptoKey(data.dhPrivateKey, ecdhParams, ['deriveKey', 'deriveBits']),
      publicKey: await jwkToCryptoKey(data.dhPublicKey, ecdhParams, []),
    },
    remotePublicKey: data.remotePublicKey ? await jwkToCryptoKey(data.remotePublicKey, ecdhParams, []) : null,
  };
}

/** 获取本地缓存的会话加密状态 */
export function getLocalSessionStatus(sessionId: string): E2eeSessionStatus {
  const raw = localStorage.getItem(SESSION_STATUS_PREFIX + sessionId);
  if (raw === 'encrypted' || raw === 'pending') return raw;
  return 'plaintext';

/** 设置本地会话加密状态 */
export function setLocalSessionStatus(sessionId: string, status: E2eeSessionStatus): void {
  localStorage.setItem(SESSION_STATUS_PREFIX + sessionId, status);
}

/**
 * 发起 E2EE 协商（主动方）
 * 流程：检查本地密钥 → 获取对方 Bundle → X3DH → 初始化 Double Ratchet → 保存状态
 */
export async function initiateNegotiation(
  sessionId: string,
  remoteUserId: string,
  remoteDeviceId?: string,
): Promise<boolean> {
  try {
    // 1. 确保本机密钥已注册
    const identityKeyPair = await getIdentityKeyPair();
    if (!identityKeyPair) {
      throw new Error('Local identity key not found. Run device registration first.');
    }

    // 2. 获取对方公钥 Bundle
    const bundleResp = await keyService.getBundle(remoteUserId, remoteDeviceId);
    if (!bundleResp.data) {
      throw new Error('Remote user has no E2EE key bundle registered');
    }
    const remoteBundle: PreKeyBundle = {
      userId: remoteUserId,
      deviceId: bundleResp.data.deviceId,
      identityKey: bundleResp.data.identityKey,
      signedPreKey: bundleResp.data.signedPreKey,
      signedPreKeySignature: bundleResp.data.signedPreKeySignature,
      oneTimePreKey: (bundleResp.data as Record<string, unknown>).oneTimePreKey as string | undefined,
    };

    // 3. X3DH 密钥协商
    const { rootKey, ephemeralPublicKey } = await x3dhInitiate(identityKeyPair, remoteBundle);

    // 4. 初始化 Double Ratchet 发送链
    const ratchetState = await initSendingChain(rootKey, identityKeyPair);

    // 5. 持久化棘轮状态
    await saveRatchetState(sessionId, ratchetState);

    // 6. 更新本地会话状态
    setLocalSessionStatus(sessionId, 'encrypted');

    return true;
  } catch (error) {
    console.error('[E2EE] Negotiation initiation failed:', error);
    return false;
  }
}

/**
 * 响应 E2EE 协商（被动方）
 * 流程：收到对方首条加密消息 → X3DH 响应 → 初始化接收状态
 */
export async function respondToNegotiation(
  sessionId: string,
  remoteUserId: string,
  remoteIdentityKeyBase64: string,
  ephemeralPublicKeyBase64: string,
): Promise<boolean> {
  try {
    const identityKeyPair = await getIdentityKeyPair();
    if (!identityKeyPair) {
      throw new Error('Local identity key not found');
    }

    // 从本地存储获取 signedPreKey 和 oneTimePreKey
    const localBundle = await savePreKeyBundle(null as any); // TODO: 读取本地 bundle
    if (!localBundle) {
      throw new Error('Local pre-key bundle not found');
    }

    // X3DH 响应
    const rootKey = await x3dhRespond(
      identityKeyPair,
      localBundle.signedPreKeyPair,
      localBundle.oneTimePreKeyPairs?.[0],
      remoteIdentityKeyBase64,
      ephemeralPublicKeyBase64,
    );

    // 初始化发送链
    const ratchetState = await initSendingChain(rootKey, identityKeyPair);
    await saveRatchetState(sessionId, ratchetState);
    setLocalSessionStatus(sessionId, 'encrypted');

    return true;
  } catch (error) {
    console.error('[E2EE] Negotiation response failed:', error);
    return false;
  }
}

/**
 * 检查并恢复 E2EE 会话
 * 用于应用重启后从 IndexedDB 恢复已有加密会话
 */
export async function restoreE2eeSession(sessionId: string): Promise<boolean> {
  const state = await getRatchetState(sessionId);
  if (!state) return false;
  setLocalSessionStatus(sessionId, 'encrypted');
  return true;
}

// frontend/src/features/e2ee/manager/e2ee-manager.ts
import { useUserStore } from '@/stores/user';
import { ratchetEncrypt, ratchetDecrypt } from '../engine/double-ratchet';
import { getRatchetState, saveRatchetState } from '../store/session-store';
import { MessageBuffer } from '../engine/message-buffer';
import { getLocalSessionStatus, initiateNegotiation } from './negotiation';
import type { RatchetHeader, E2eeSessionStatus } from '../types';

interface EncryptedPayload {
  ciphertext: string;
  header: RatchetHeader;
  deviceId: string;
}

interface PendingDecrypt {
  sessionId: string;
  header: RatchetHeader;
  ciphertext: string;
  resolve: (plaintext: string) => void;
  reject: (error: Error) => void;
}

class E2eeManager {
  private buffer = new MessageBuffer();
  private pendingQueue: PendingDecrypt[] = [];
  private deviceId: string = '';

  /** 初始化（应用启动时调用） */
  async init(deviceId: string): Promise<void> {
    this.deviceId = deviceId;
  }

  /** 获取会话加密状态 */
  getSessionStatus(sessionId: string): E2eeSessionStatus {
    return getLocalSessionStatus(sessionId);
  }

  /** 加密消息（私聊） */
  async encryptMessage(sessionId: string, plaintext: string): Promise<EncryptedPayload | null> {
    const state = await getRatchetState(sessionId);
    if (!state) {
      console.warn('[E2EE] No ratchet state for session, falling back to plaintext');
      return null;
    }

    const { header, ciphertext } = await ratchetEncrypt(state, plaintext);
    await saveRatchetState(sessionId, state);

    return { ciphertext, header, deviceId: this.deviceId };
  }

  /** 解密消息（私聊） */
  async decryptMessage(
    sessionId: string,
    senderId: string,
    header: RatchetHeader,
    ciphertext: string,
  ): Promise<string> {
    const state = await getRatchetState(sessionId);
    if (!state) {
      throw new Error(`No ratchet state for session ${sessionId}`);
    }

    const result = this.buffer.enqueue(
      header.counter,
      header,
      ciphertext,
      state.receiveCounter,
    );

    if (result.action === 'drop') {
      console.warn('[E2EE] Dropping duplicate/out-of-order message');
      return '';
    }

    if (result.action === 'renegotiate') {
      console.warn('[E2EE] Counter gap too large, renegotiation needed');
      await initiateNegotiation(sessionId, senderId);
      throw new Error('Session renegotiation required');
    }

    if (result.action === 'buffer') {
      // 消息已缓冲，等待后续消息补齐
      return '';
    }

    // action === 'decrypt'，按序解密
    const decryptedParts: string[] = [];
    for (const msg of result.messages) {
      const plaintext = await ratchetDecrypt(state, msg.header, msg.ciphertext);
      decryptedParts.push(plaintext);
    }
    await saveRatchetState(sessionId, state);

    return decryptedParts.join('');
  }

  /** 清除会话加密状态 */
  async clearSession(sessionId: string): Promise<void> {
    this.buffer.clear();
    localStorage.removeItem('e2ee:status:' + sessionId);
  }
}

export const e2eeManager = new E2eeManager();

// frontend/src/features/e2ee/engine/sender-key.ts
import {
  generateSigningKeyPair,
  exportPublicKey,
  importPublicKey,
  hkdfDeriveKey,
  aesGcmEncrypt,
  aesGcmDecrypt,
  ecdsaSign,
  ecdsaVerify,
} from './crypto-primitives';
import { bufferToBase64, base64ToBuffer, randomBytes } from './codec';

export interface SenderKey {
  chainKey: CryptoKey;
  signingKeyPair: CryptoKeyPair;
  counter: number;
}

export interface SenderKeyHeader {
  signingPubkey: string;  // Base64
  counter: number;
  signature: string;      // Base64
  iv: string;             // Base64
}

/** 生成新的 Sender Key（管理员开启群加密时调用） */
export async function generateSenderKey(): Promise<SenderKey> {
  const chainKeyRaw = randomBytes(32);
  const chainKey = await crypto.subtle.importKey(
    'raw',
    chainKeyRaw,
    { name: 'HKDF' },
    false,
    ['deriveKey', 'deriveBits'],
  );
  const signingKeyPair = await generateSigningKeyPair();
  return { chainKey, signingKeyPair, counter: 0 };
}

/** 从链密钥派生消息密钥和下一个链密钥 */
async function splitSenderChainKey(chainKey: CryptoKey): Promise<{
  messageKey: CryptoKey;
  nextChainKey: CryptoKey;
}> {
  const raw = await crypto.subtle.exportKey('raw', chainKey);
  const salt = new Uint8Array(0).buffer;
  const mkInfo = new TextEncoder().encode('SenderKeyMessageKeys').buffer;
  const ckInfo = new TextEncoder().encode('SenderKeyChainKeys').buffer;
  const messageKey = await hkdfDeriveKey(raw, salt, mkInfo);
  const nextChainKey = await hkdfDeriveKey(raw, salt, ckInfo);
  return { messageKey, nextChainKey };
}

/** 用 Sender Key 加密消息（发送方调用一次） */
export async function senderKeyEncrypt(
  senderKey: SenderKey,
  plaintext: string,
): Promise<{ ciphertext: string; header: SenderKeyHeader }> {
  const { messageKey, nextChainKey } = await splitSenderChainKey(senderKey.chainKey);
  senderKey.chainKey = nextChainKey;

  const iv = randomBytes(12);
  const ptBuffer = new TextEncoder().encode(plaintext).buffer;
  const { ciphertext } = await aesGcmEncrypt(messageKey, ptBuffer, iv);

  // 生成签名：sign(ciphertext || counter)
  const counterBytes = new Uint8Array(4);
  new DataView(counterBytes.buffer).setUint32(0, senderKey.counter, false);
  const signData = new Uint8Array([
    ...new Uint8Array(ciphertext),
    ...counterBytes,
  ]).buffer;
  const signature = await ecdsaSign(senderKey.signingKeyPair.privateKey, signData);

  const signingPubRaw = await exportPublicKey(senderKey.signingKeyPair.publicKey);

  const header: SenderKeyHeader = {
    signingPubkey: bufferToBase64(signingPubRaw),
    counter: senderKey.counter,
    signature: bufferToBase64(signature),
    iv: bufferToBase64(iv.buffer),
  };

  senderKey.counter++;
  return { ciphertext: bufferToBase64(ciphertext), header };
}

/** 用 Sender Key 解密消息（接收方调用） */
export async function senderKeyDecrypt(
  senderKey: SenderKey,
  header: SenderKeyHeader,
  ciphertextBase64: string,
): Promise<string> {
  const signingPub = await importPublicKey(base64ToBuffer(header.signingPubkey));
  const ciphertext = base64ToBuffer(ciphertextBase64);
  const iv = new Uint8Array(base64ToBuffer(header.iv));

  // 验证签名
  const counterBytes = new Uint8Array(4);
  new DataView(counterBytes.buffer).setUint32(0, header.counter, false);
  const signData = new Uint8Array([
    ...new Uint8Array(ciphertext),
    ...counterBytes,
  ]).buffer;
  const valid = await ecdsaVerify(signingPub, base64ToBuffer(header.signature), signData);
  if (!valid) throw new Error('Sender Key signature verification failed');

  // 推进链密钥到目标 counter
  let currentChainKey = senderKey.chainKey;
  for (let i = senderKey.counter; i <= header.counter; i++) {
    const { messageKey, nextChainKey } = await splitSenderChainKey(currentChainKey);
    if (i === header.counter) {
      const decrypted = await aesGcmDecrypt(messageKey, ciphertext, iv);
      senderKey.chainKey = nextChainKey;
      senderKey.counter = header.counter + 1;
      return new TextDecoder().decode(decrypted);
    }
    currentChainKey = nextChainKey;
  }
  throw new Error('Sender Key counter mismatch');
}

/** 序列化 Sender Key 用于加密传输（用接收者公钥加密） */
export async function serializeSenderKey(senderKey: SenderKey): Promise<{
  chainKeyRaw: ArrayBuffer;
  signingPubRaw: ArrayBuffer;
  signingPrivJwk: JsonWebKey;
}> {
  const chainKeyRaw = await crypto.subtle.exportKey('raw', senderKey.chainKey);
  const signingPubRaw = await exportPublicKey(senderKey.signingKeyPair.publicKey);
  const signingPrivJwk = await crypto.subtle.exportKey('jwk', senderKey.signingKeyPair.privateKey);
  return { chainKeyRaw, signingPubRaw, signingPrivJwk };
}

/** 反序列化 Sender Key */
export async function deserializeSenderKey(data: {
  chainKeyRaw: ArrayBuffer;
  signingPubRaw: ArrayBuffer;
  signingPrivJwk: JsonWebKey;
}): Promise<SenderKey> {
  const chainKey = await crypto.subtle.importKey(
    'raw', data.chainKeyRaw, { name: 'HKDF' }, false, ['deriveKey', 'deriveBits'],
  );
  const signingPub = await importPublicKey(data.signingPubRaw);
  const signingPriv = await crypto.subtle.importKey(
    'jwk', data.signingPrivJwk,
    { name: 'ECDSA', namedCurve: 'P-256' },
    false, ['sign'],
  );
  return {
    chainKey,
    signingKeyPair: { privateKey: signingPriv, publicKey: signingPub },
    counter: 0,
  };
}

const CHUNK_SIZE = 5 * 1024 * 1024; // 5MB per chunk

  mediaKey: string;        // Base64
  chunkIvs: string[];      // Base64 per chunk
  mimeType: string;

export async function encryptMedia(file: File): Promise<EncryptedMediaResult> {
  if (file.size > LARGE_FILE_THRESHOLD) {

  }

  const mediaKeyRaw = randomBytes(32);
  const mediaKey = await crypto.subtle.importKey(
    'raw', mediaKeyRaw, { name: 'AES-GCM', length: 256 }, false, ['encrypt'],
  );

  const fileBuffer = await file.arrayBuffer();
  const iv = randomBytes(12);
  const ciphertext = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    mediaKey,
    fileBuffer,
  );

  return {
    encryptedChunks: [new Blob([ciphertext])],
    mediaKey: btoa(String.fromCharCode(...new Uint8Array(mediaKeyRaw))),
    chunkIvs: [btoa(String.fromCharCode(...iv))],
  };
}

/** Worker 加密（>=5MB 文件） */
async function encryptMediaWithWorker(file: File): Promise<EncryptedMediaResult> {
  const mediaKeyRaw = randomBytes(32);
  const mediaKeyBase64 = btoa(String.fromCharCode(...new Uint8Array(mediaKeyRaw)));

  const fileBuffer = await file.arrayBuffer();
  const totalChunks = Math.ceil(fileBuffer.byteLength / CHUNK_SIZE);

  const worker = new Worker(
    new URL('../workers/media-crypto.worker.ts', import.meta.url),
    { type: 'module' },
  );

  return new Promise((resolve, reject) => {
    const encryptedChunks: Blob[] = [];
    const chunkIvs: string[] = [];
    let processedChunks = 0;

    worker.onmessage = (e: MessageEvent) => {
      const { type, chunkIndex, encryptedChunk, iv, error } = e.data;
      if (type === 'error') {
        worker.terminate();
        reject(new Error(error));
        return;
      }
      if (type === 'chunk-done') {
        encryptedChunks[chunkIndex] = new Blob([encryptedChunk]);
        chunkIvs[chunkIndex] = iv;
        processedChunks++;
        if (processedChunks === totalChunks) {
          worker.terminate();
          resolve({ encryptedChunks, mediaKey: mediaKeyBase64, chunkIvs });
        }
      }
    };

    worker.onerror = (e) => {
      worker.terminate();
      reject(e.error || new Error('Worker error'));
    };

    // 发送加密任务给 Worker
    worker.postMessage({
      type: 'encrypt',
      fileBuffer,
      mediaKeyBase64,
      chunkSize: CHUNK_SIZE,
    });
  });
}

/** 解密媒体文件 */
export async function decryptMedia(params: MediaDecryptParams): Promise<Blob> {
  const mediaKeyRaw = Uint8Array.from(atob(params.mediaKey), c => c.charCodeAt(0));
  const mediaKey = await crypto.subtle.importKey(
    'raw', mediaKeyRaw, { name: 'AES-GCM', length: 256 }, false, ['decrypt'],
  );

  const decryptedParts: ArrayBuffer[] = [];
  for (let i = 0; i < params.encryptedChunks.length; i++) {
    const chunkBuffer = await params.encryptedChunks[i].arrayBuffer();
    const iv = Uint8Array.from(atob(params.chunkIvs[i]), c => c.charCodeAt(0));
    const decrypted = await crypto.subtle.decrypt(
      { name: 'AES-GCM', iv },
      mediaKey,
      chunkBuffer,
    );
    decryptedParts.push(decrypted);
  }

  return new Blob(decryptedParts, { type: params.mimeType });
}

// frontend/src/features/e2ee/workers/media-crypto.worker.ts

interface EncryptMessage {
  type: 'encrypt';
  fileBuffer: ArrayBuffer;
  mediaKeyBase64: string;
  chunkSize: number;
}

self.onmessage = async (e: MessageEvent<EncryptMessage>) => {
  const { fileBuffer, mediaKeyBase64, chunkSize } = e.data;

  try {
    const mediaKeyRaw = Uint8Array.from(atob(mediaKeyBase64), c => c.charCodeAt(0));
    const mediaKey = await crypto.subtle.importKey(
      'raw', mediaKeyRaw, { name: 'AES-GCM', length: 256 }, false, ['encrypt'],
    );

    const totalChunks = Math.ceil(fileBuffer.byteLength / chunkSize);

    for (let i = 0; i < totalChunks; i++) {
      const start = i * chunkSize;
      const end = Math.min(start + chunkSize, fileBuffer.byteLength);
      const chunk = fileBuffer.slice(start, end);

      const iv = crypto.getRandomValues(new Uint8Array(12));
      const encryptedChunk = await crypto.subtle.encrypt(
        { name: 'AES-GCM', iv },
        mediaKey,
        chunk,
      );

      const ivBase64 = btoa(String.fromCharCode(...iv));
      self.postMessage({
        type: 'chunk-done',
        chunkIndex: i,
        encryptedChunk,
        iv: ivBase64,
      }, [encryptedChunk]);
    }
  } catch (error) {
    self.postMessage({
      type: 'error',
      error: error instanceof Error ? error.message : String(error),
    });
  }
};

  agreed.value = false;

const handleConfirm = async () => {
  loading.value = true;
  try {
    // 为每个成员的每台设备生成加密的 Sender Key
    const { generateSenderKey, serializeSenderKey } = await import('@/features/e2ee/engine/sender-key');
    const { importPublicKey, ecdhDeriveBits } = await import('@/features/e2ee/engine/crypto-primitives');
    const { base64ToBuffer, bufferToBase64 } = await import('@/features/e2ee/engine/codec');
    const { getIdentityKeyPair } = await import('@/features/e2ee/store/key-store');

    const myKeyPair = await getIdentityKeyPair();
    if (!myKeyPair) throw new Error('Local identity key not found');

    const senderKey = await generateSenderKey();
    const serialized = await serializeSenderKey(senderKey);

    // 用每个成员的公钥加密 Sender Key

          deviceId: device.deviceId,
