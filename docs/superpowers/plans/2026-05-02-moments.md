# Moments (朋友圈) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement WeChat-style Moments social feed feature for NewIM platform.

**Architecture:** MySQL persistence + Redis Feed cache + WebSocket real-time push. Posts fan-out to friends' feed caches asynchronously.

**Tech Stack:** Rust (api-server-rs), Vue 3 + TypeScript + Pinia, MySQL 8, Redis, WebSocket

```sql
USE service_message_service_db;

-- 动态表
CREATE TABLE IF NOT EXISTS moments_post (
    id BIGINT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    content TEXT,
    visibility TINYINT NOT NULL DEFAULT 0 COMMENT '0=公开, 1=好友可见, 2=仅自己可见',
    link_url VARCHAR(512),
    link_title VARCHAR(256),
    link_cover VARCHAR(512),
    location VARCHAR(255),
    status TINYINT NOT NULL DEFAULT 0 COMMENT '0=正常, 1=已删除',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_user_id_created (user_id, created_at DESC),
    INDEX idx_created (created_at DESC)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- 媒体资源表
CREATE TABLE IF NOT EXISTS moments_media (
    id BIGINT PRIMARY KEY,
    post_id BIGINT NOT NULL,
    type TINYINT NOT NULL COMMENT '0=图片, 1=视频',
    url VARCHAR(512) NOT NULL,
    sort_order TINYINT NOT NULL DEFAULT 0,
    INDEX idx_post_id (post_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- 点赞表
CREATE TABLE IF NOT EXISTS moments_like (
    id BIGINT PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_post_user (post_id, user_id),
    INDEX idx_post_id (post_id),
    INDEX idx_user_id (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- 评论表
CREATE TABLE IF NOT EXISTS moments_comment (
    id BIGINT PRIMARY KEY,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    parent_id BIGINT COMMENT 'NULL=顶级评论',
    content TEXT NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_post_id_created (post_id, created_at),
    INDEX idx_parent_id (parent_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
```

```rust
use actix_web::{web, HttpRequest, HttpResponse};
use serde::{Deserialize, Serialize};
use sqlx::MySqlPool;

use crate::error::AppError;
use crate::auth::Claims;

#[derive(Debug, Deserialize)]
pub struct CreatePostRequest {
    pub content: Option<String>,
    pub visibility: Option<i8>,
    pub link_url: Option<String>,
    pub link_title: Option<String>,
    pub link_cover: Option<String>,
    pub location: Option<String>,
}

#[derive(Debug, Deserialize)]
pub struct FeedQuery {
    pub cursor: Option<i64>,
    pub limit: Option<i64>,
}

pub async fn create_post(
    req: HttpRequest,
    pool: web::Data<MySqlPool>,
    form: web::Json<CreatePostRequest>,
) -> Result<HttpResponse, AppError> {
    let claims = Claims::from_request(&req)?;
    let user_id = claims.user_id;
    let post_id = im_rs_common::ids::next_id();

    let visibility = form.visibility.unwrap_or(0);
    let now = chrono::Utc::now().naive_utc();

    sqlx::query!(
        r#"
        INSERT INTO moments_post (id, user_id, content, visibility, link_url, link_title, link_cover, location, status, created_at, updated_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, 0, ?, ?)
        "#,
        post_id,
        user_id,
        form.content,
        visibility,
        form.link_url,
        form.link_title,
        form.link_cover,
        form.location,
        now,
        now
    )
    .execute(pool.get_ref())
    .await?;

    // TODO: Trigger async fan-out to friends' feed caches

    Ok(HttpResponse::Ok().json(serde_json::json!({
        "success": true,
        "data": { "id": post_id }
    })))
}

pub async fn get_feed(
    req: HttpRequest,
    pool: web::Data<MySqlPool>,
    query: web::Query<FeedQuery>,
) -> Result<HttpResponse, AppError> {
    let claims = Claims::from_request(&req)?;
    let user_id = claims.user_id;
    let cursor = query.cursor.unwrap_or(i64::MAX);
    let limit = query.limit.unwrap_or(20).min(50);

    // TODO: Check Redis cache first, fallback to MySQL

    let posts = sqlx::query!(
        r#"
        SELECT p.id, p.user_id, p.content, p.visibility, p.link_url, p.link_title, p.link_cover, p.location, p.status, p.created_at, p.updated_at
        FROM moments_post p
        WHERE p.id < ? AND p.status = 0
        ORDER BY p.id DESC
        LIMIT ?
        "#,
        cursor,
        limit
    )
    .fetch_all(pool.get_ref())
    .await?;

    Ok(HttpResponse::Ok().json(serde_json::json!({
        "success": true,
        "data": posts
    })))
}

pub async fn get_post(
    req: HttpRequest,
    pool: web::Data<MySqlPool>,
    path: web::Path<i64>,
) -> Result<HttpResponse, AppError> {
    let claims = Claims::from_request(&req)?;
    let post_id = path.into_inner();

    let post = sqlx::query!(
        r#"
        SELECT id, user_id, content, visibility, link_url, link_title, link_cover, location, status, created_at, updated_at
        FROM moments_post
        WHERE id = ? AND status = 0
        "#,
        post_id
    )
    .fetch_optional(pool.get_ref())
    .await?;

    match post {
        Some(p) => Ok(HttpResponse::Ok().json(serde_json::json!({
            "success": true,
            "data": p
        }))),
        None => Err(AppError::NotFound("Post not found".into())),
    }
}
```

```rust
use redis::aio::ConnectionManager;
use redis::AsyncCommands;

use im_rs_common::keys;

pub async fn add_to_feed(
    conn: &mut ConnectionManager,
    user_id: i64,
    post_id: i64,
) -> Result<(), redis::RedisError> {
    let key = format!("{}{}", keys::MOMENTS_FEED_PREFIX, user_id);
    conn.zadd::<_, _, _, ()>(&key, post_id, post_id).await?;
    conn.expire(&key, keys::MOMENTS_FEED_TTL as usize).await?;
    Ok(())
}

pub async fn get_feed(
    conn: &mut ConnectionManager,
    user_id: i64,
    cursor: i64,
    limit: i64,
) -> Result<Vec<i64>, redis::RedisError> {
    let key = format!("{}{}", keys::MOMENTS_FEED_PREFIX, user_id);
    let post_ids: Vec<i64> = conn.zrevrangebyscore(&key, cursor - 1, "-inf").await?;
    Ok(post_ids.into_iter().take(limit as usize).collect())
}

pub async fn cache_post(
    conn: &mut ConnectionManager,
    post_id: i64,
    post_json: &str,
) -> Result<(), redis::RedisError> {
    let key = format!("{}{}", keys::MOMENTS_POST_PREFIX, post_id);
    conn.set_ex::<_, _, ()>(&key, post_json, keys::MOMENTS_POST_TTL as usize).await?;
    Ok(())
}

pub async fn get_cached_post(
    conn: &mut ConnectionManager,
    post_id: i64,
) -> Result<Option<String>, redis::RedisError> {
    let key = format!("{}{}", keys::MOMENTS_POST_PREFIX, post_id);
    conn.get(&key).await
}

pub async fn invalidate_post(
    conn: &mut ConnectionManager,
    post_id: i64,
) -> Result<(), redis::RedisError> {
    let key = format!("{}{}", keys::MOMENTS_POST_PREFIX, post_id);
    conn.del(&key).await?;
    Ok(())
}

pub async fn add_notification(
    conn: &mut ConnectionManager,
    user_id: i64,
    notification_id: i64,
) -> Result<(), redis::RedisError> {
    let key = format!("{}{}", keys::MOMENTS_NOTIFY_PREFIX, user_id);
    conn.zadd::<_, _, _, ()>(&key, notification_id, chrono::Utc::now().timestamp()).await?;
    conn.expire(&key, keys::MOMENTS_NOTIFY_TTL as usize).await?;
    Ok(())
}

pub async fn get_unread_count(
    conn: &mut ConnectionManager,
    user_id: i64,
) -> Result<i64, redis::RedisError> {
    let key = format!("{}{}:unread", keys::MOMENTS_NOTIFY_PREFIX, user_id);
    conn.get(&key).await.unwrap_or(Ok(0))
}
```

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { momentsService } from '@/services/moments'
import type { PostWithDetails, MomentNotification } from '@/types/moments'

export const useMomentsStore = defineStore('moments', () => {
  // State
  const feed = ref<PostWithDetails[]>([])
  const notifications = ref<MomentNotification[]>([])
  const unreadCount = ref(0)
  const loading = ref(false)
  const hasMore = ref(true)

  // Getters
  const sortedFeed = computed(() => [...feed.value].sort((a, b) =>
    Number(BigInt(b.post.id) - BigInt(a.post.id))
  ))

  // Actions
  async function loadFeed(refresh = false) {
    if (loading.value) return
    if (!refresh && !hasMore.value) return

    loading.value = true
    try {
      const cursor = refresh ? undefined : feed.value[feed.value.length - 1]?.post.id
      const newPosts = await momentsService.getFeed({ cursor, limit: 20 })

      if (refresh) {
        feed.value = newPosts
      } else {
        feed.value.push(...newPosts)
      }

      hasMore.value = newPosts.length === 20
    } finally {
      loading.value = false
    }
  }

  async function addPost(post: PostWithDetails) {
    feed.value.unshift(post)
  }

  async function removePost(postId: string) {
    feed.value = feed.value.filter(p => p.post.id !== postId)
  }

  async function toggleLike(postId: string) {
    const post = feed.value.find(p => p.post.id === postId)
    if (!post) return

    if (post.isLiked) {
      await momentsService.unlikePost(postId)
      post.isLiked = false
      post.likeCount--
    } else {
      await momentsService.likePost(postId)
      post.isLiked = true
      post.likeCount++
    }
  }

  async function loadNotifications() {
    notifications.value = await momentsService.getNotifications()
    unreadCount.value = notifications.value.filter(n => !n.isRead).length
  }

  async function markNotificationsRead() {
    await momentsService.markNotificationsRead()
    notifications.value.forEach(n => n.isRead = true)
    unreadCount.value = 0
  }

  return {
    feed,
    notifications,
    unreadCount,
    loading,
    hasMore,
    sortedFeed,
    loadFeed,
    addPost,
    removePost,
    toggleLike,
    loadNotifications,
    markNotificationsRead
  }
})
```

```typescript
import { ref } from 'vue'
import { momentsService } from '@/services/moments'
import type { MomentLike, MomentComment } from '@/types/moments'

export function useMomentsInteractions(postId: string) {
  const likes = ref<MomentLike[]>([])
  const comments = ref<MomentComment[]>([])
  const loadingLikes = ref(false)
  const loadingComments = ref(false)

  async function loadLikes() {
    loadingLikes.value = true
    try {
      likes.value = await momentsService.getLikes(postId)
    } finally {
      loadingLikes.value = false
    }
  }

  async function loadComments() {
    loadingComments.value = true
    try {
      comments.value = await momentsService.getComments(postId)
    } finally {
      loadingComments.value = false
    }
  }

  async function addComment(content: string, parentId?: string) {
    const result = await momentsService.createComment(postId, { content, parentId })
    const newComment = await momentsService.getComments(postId)
    comments.value = newComment
    return result
  }

  async function removeComment(commentId: string) {
    await momentsService.deleteComment(commentId)
    comments.value = comments.value.filter(c => c.id !== commentId)
  }

  return {
    likes,
    comments,
    loadingLikes,
    loadingComments,
    loadLikes,
    loadComments,
    addComment,
    removeComment
  }
}
```
